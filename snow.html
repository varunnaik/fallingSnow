Options and config

            var snow = fallingSnow();
            snow.fall(
                speed: 0.5, // 0.3-1.0, 1.0 = faster
                intensity: 10, // 10-100, 100 = blizzard
                snowflakeSize: 3, // 1-5, 5 = huge flakes
                element: 'html' // QuerySelector used to add flakes to an element        
            );
